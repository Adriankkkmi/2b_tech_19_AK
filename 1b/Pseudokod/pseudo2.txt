-- pętla for

-- 1. Napisz program, który oblicza sumę liczb dwucyfrowych

suma = 0
for i in range(10,100):
    suma = suma + i
print(suma)

-- Pseudokod

suma = 0
dla każdego i=11,12,13,...,99 wykonuj
    suma = suma + 1
wypisz suma


-- 2. Napisz program, który obliczy sumę k liczb wpisanych przez usera

k = int(input())
suma = 0
for i in range(k):
    x = int(input())
    suma = suma + x
print(suma)

-- Pseudokod

czytaj k
suma = 0
dla każdego i=1,2,...k wykonuj
    czytaj x
    suma = suma + x
wypisz suma
